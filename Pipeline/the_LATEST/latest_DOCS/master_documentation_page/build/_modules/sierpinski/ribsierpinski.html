[docs]def make_sierpinski_points_from_obj(name, reps=1000, seed=None):
    """
    Creates a number of points from a base object in the form of a sierpinski
    gasket

    Args:
        name (str): The name of the object that you wish to use as a base
        .. Note::
         Using a tetrahedron (pyramid) or a cube is recommended
        reps (int): The number of points for the current function to make
        seed (int): For consistent psuedo-random generation, set to a number.
                    If nothing is specified, a random result is generated
                    each time

    Returns:
        tuple: Contains the list of points in the form of a generator and
               dictionary with the min/max bounds of the base object

    """
    if reps < 4:
        sys.exit("Reps must be greater than 4")

    if seed is not None:
        random.seed(seed)  # pseudo-random generation
    # generate sierpinski gasket data
    seed = [0, 0, 0]

    try:
        tetra = pm.PyNode(name)
    except pm.MayaNodeError:
        tetra = pm.polyPyramid()[0]
    finally:
        verts = tetra.getShape().getPoints(space="world")
        bb = tetra.getBoundingBoxInvisible()
        bbMinX = bb[0]
        bbMinY = bb[1]
        bbMinZ = bb[2]
        bbMaxX = bb[3]
        bbMaxY = bb[4]
        bbMaxZ = bb[5]

    pointList = sierpin.sierpinski(seed, verts, reps)
    outputDict = (pointList, {'minX': bbMinX,
                              'minY': bbMinY,
                              'minZ': bbMinZ,
                              'maxX': bbMaxX,
                              'maxY': bbMaxY,
                              'maxZ': bbMaxZ})
    return outputDict

# end make_sierpinski_points_from_obj



[docs]def points_to_rib_write(name, reps, sierpinski, templateLoc, outputLoc):
    """
    Creates sierpinski gasket points and then writes it to a rib file

    Args:
        name (str): Name of the object to use as the bounds for sierpinski
                    gasket
        reps (int): The number of points to create
        sierpinski (tuple): The sierpinki data to pass through
        templateLoc (str): To simplify the act of writing a rib file with highly
                           standardized info, simply create a template file with
                           formatted text to replace with point values using {}s
        outputLoc (str): The location of where to write the final rib file to

    Returns:
        True or NoneType: Returns the success of the function
    """
    if sierpinski is None:
        sierpinski, bb = make_sierpinski_points_from_obj(name, reps)
    sierpinski, bb = sierpinski
    pointList = [str(item) for sublist in list(sierpinski) for item in sublist]
    pointList = ' '.join(pointList)

    # get file and write
    ribOut = ribparticle.RibOutput()
    with open(templateLoc) as f:
        data = f.readlines()
    data = ''.join(data)

    outputData = data.format(points=pointList, constwidth=0.01,
                             minX=bb['minX'],
                             minY=bb['minY'],
                             minZ=bb['minZ'],
                             maxX=bb['maxX'],
                             maxY=bb['maxY'],
                             maxZ=bb['maxZ'])
    with open(outputLoc, 'w') as f:
        f.write(outputData)

    return True

    templateDir = os.path.join(rootPath, "rib_template.rib")
    if not os.path.isdir(outputDir):
        sys.exit("Dir: {} does not exist".format(outputDir))
        paths.mkdir_p(outputDir)

    framePadding = len(str(endTime)) + 1
    timeRange = endTime - startTime
    for f in xrange(timeRange + 1):
        filename = 'tetra_out_{}.rib'.format(str(startTime + f).zfill(framePadding))

        pm.currentTime(startTime + f)
        sierpinski = make_sierpinski_points_from_obj(name, reps)
