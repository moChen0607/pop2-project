[docs]def midpoint_3d(p1, p2):
    """
    Gets the midpoint of two iterables in an efficient one liner. The
    function is just for syntax sugar

    .. Considerations:
     p1 and p2 have their indices iterated through in order for the funciton to
     work. Because of this, p1 and p2 must be of equal length

    Args:
        p1 (list): A list or iterable of points to get the midpoint from
        p2 (list): A list or iterable of points to get the midpoint from

    Returns:
        list or iterable: The output midpoint between the two points
    """

    if len(p1) != len(p2):
        raise ValueError('p1 "{}" and p2 "{}" are not equal length'.format(p1, p2))

    midpoint = [(value + p2[index]) / 2 for index, value in enumerate(p1)]
    return midpoint

# end midpoint_3d

    return pyPyramidShape

[docs]def sierpinski(seed, verts, reps):
    """
    Creates sierpinski pattern from a at least 3 points

    Args:
        seed (list): Any 3D point. [0, 0, 0] is fine in most cases
        verts (list of lists): A list containing 3D position points in world
                               space. Represents every point in a base
                               tetrahedron
        reps: (int): The number of points to generate from the sierpinski

    Returns:
        generator: The created points from the sierpinski algorithm
    """
    if len(verts) < 4:
        sys.exit("Please provide at least 4 points")

    for n in xrange(reps):
        randPoint = random.choice(verts)
        midPnt = [(value + seed[index]) / 2 for index, value in enumerate(randPoint)]
        seed = midPnt
        yield midPnt

# end sierpinski
