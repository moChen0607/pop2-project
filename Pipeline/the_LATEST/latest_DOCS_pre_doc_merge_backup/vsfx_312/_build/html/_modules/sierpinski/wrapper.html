# -*- coding: utf-8 -*-

# IMPORT THIRD-PARTY LIBRARIES
import maya.cmds as cmds
import pymel.core as pm



[docs]def wrap_objects(wrapper, wrapperCtl, selection):
    """
    Wrap objects to a single wrapper object and creates a couple interface
    controls all of the wrapped objects (so that the wrap can be toggled
    on/off more easily)

    .. Note::
     I had to dig quite a bit to find CreateWrap, which actually isn't a pymel
     command, which is why the call uses cmds. Also, the command returns None,
     even on success, so I had to manually refind the wrap node that was created
     by searching the selection object's history. Thankfully history is listed
     most-recent to oldest so I'm able to get away with that without
     accidentally picking up a wrap node that I didn't intend to

    Args:
        wrapper (<pymel.core.nodetypes.Mesh>): The shape to wrap the selection to

        wrapperCtl (<pymel.core.nodetypes.Transform>): Convenience controller

        selection (<pymel.core.nodetypes.Transform>): A iterable of object transforms

    Returns:
        NoneType: None
    """
    # initial settings
    try:
        wrapValue = pm.getAttr(wrapperCtl + ".enableWrap")
    except pm.MayaAttributeError:
        pm.addAttr(wrapperCtl, ln="enableWrap", at="bool")
        wrapValue = pm.getAttr(wrapperCtl + ".enableWrap")
    pm.setAttr(wrapperCtl + ".enableWrap", False)  # speed up script a bit
    for sel in selection:
        pm.select(sel, wrapper, r=True)
        cmds.CreateWrap(sel.getShape().nodeName(), wrapper.nodeName())
        for hist in sel.listHistory():
            if isinstance(hist, pm.nodetypes.Wrap):
                wrapperDeformer = hist
                break  # found the most recent wrapper
        pm.connectAttr(wrapperCtl + ".enableWrap", wrapperDeformer.envelope)

    # restore initial settings
    pm.setAttr(wrapperCtl + ".enableWrap", wrapValue)

# end wrap_objects


def main():
    # get pynodes
    wrapper = 'tetra_low_geo'  # the geometry used to make the base sierpinski
    wrapper = pm.PyNode(wrapper)
    wrapperCtl = 'tetra_CTL'  # name of the control curve to add attributes to
    wrapperCtl = pm.PyNode(wrapperCtl)

    # get original selection
    selection = pm.ls(sl=True)
    wrap_objects(wrapper, wrapperCtl)

    # reset the original selection
    pm.select(selection, r=True)
# end main


if __name__ == "__main__":
    main()
