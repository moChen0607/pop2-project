[docs]def get_output_name(selectedNodes):
    """
    Gets the output name for the finalized asset(s) in the current Nuke scene/shot
    """
    try:
        some_object_iterator = iter(selectedNodes)
        writeNode = [x for x in selectedNodes if x.Class().lower() == "write"]
        if len(writeNode) != 1:
            raise ValueError("Please select only one write node at a time to finalize")
        else:
            writeNode = writeNode[0]
    except TypeError, te:
        writeNode = selectedNodes
    rootDir = consts.FINAL_FOOTAGE_DIR

    currentSceneSave = nuke.root().name()
    currentSceneSaveSplit = paths.os_path_split_asunder(currentSceneSave)
    sceneName = currentSceneSaveSplit[-1]
    if sceneName.strip() == "" or sceneName.strip().lower() == "root":
        raise ValueError("finalize_shot - No scene name found!")
    reMatch = consts.SHOT_MATCH_PATTERN
    reCompile = re.compile(reMatch, re.VERBOSE)

    shotName = re.match(reCompile, sceneName)
    if shotName is not None:
        shotName = shotName.group()
    shotName = shotName.rstrip("_")  # remove the _ in the group

    foundSceneName = False
    for folder in currentSceneSaveSplit:
        reMatch = re.match(reCompile, folder)
        if reMatch is not None and shotName is not None and reMatch.group() == shotName:
            foundSceneName = True
        if shotName is None and reMatch is not None:
            foundSceneName = True
        if foundSceneName:
            break
    else:
        # was not found
        raise RuntimeError("Can't automatically find the scene/shot name. "
                           "Please either include it in the filename or as a "
                           "parent directory")

    writeNodeExtension = writeNode["file_type"].value()
    if writeNodeExtension.strip() == "targa":
        # The only file extension of the write node that is named differently
        # from its actual extension type must be converted manually
        #
        writeNodeExtension = "tga"

    outputPath = shotName + "_" + str(nuke.frame()).zfill(4)
    if writeNodeExtension != "":
        outputPath += "." + writeNodeExtension
    return outputPath
